  <script>

const cropForm =
document.getElementById("cropForm");

const cropContainer =
document.getElementById("cropContainer");


// LOAD CROPS
async function loadCrops(){

    const response = await fetch(
        "http://127.0.0.1:5000/api/crops"
    );

    const data = await response.json();

    cropContainer.innerHTML = "";

    data.crops.forEach((crop)=>{

        const card =
        document.createElement("div");

        card.className =
        "bg-white rounded-[30px] overflow-hidden shadow-xl";

        card.innerHTML = `

        <div class="relative aspect-square overflow-hidden">

          <img
            src="${crop.cropImage}"
            class="w-full h-full object-cover"
          >

        </div>

        <div class="p-6">

          <h3 class="font-bold text-2xl mb-2">
            ${crop.cropName}
          </h3>

          <p class="text-gray-500 mb-3">
            Farmer : ${crop.farmerName}
          </p>

          <p>🌾 ${crop.cropQty}</p>

          <p>📍 ${crop.cropVillage}</p>

          <p>🪪 ${crop.farmerPass}</p>

          <h4 class="font-bold text-xl mt-3 text-green-700">
            ${crop.cropPrice}
          </h4>

        </div>

        `;

        cropContainer.appendChild(card);

    });

}

loadCrops();


// UPLOAD CROP
cropForm.addEventListener(
    "submit",
    async function(e){

    e.preventDefault();

    const cropData = {

        farmerPass:
        document.getElementById("farmerPass").value,

        farmerName:
        document.getElementById("farmerName").value,

        cropName:
        document.getElementById("cropName").value,

        cropQty:
        document.getElementById("cropQty").value,

        cropPrice:
        document.getElementById("cropPrice").value,

        cropVillage:
        document.getElementById("cropVillage").value,

        cropImage:
        document.getElementById("cropImage").value

    };

    const response = await fetch(
        "http://127.0.0.1:5000/api/upload-crop",
        {

            method:"POST",

            headers:{
                "Content-Type":"application/json"
            },

            credentials:"include",

            body:JSON.stringify(cropData)

        }
    );

    const data = await response.json();

    alert(data.message);

    if(data.success){

        cropForm.reset();

        loadCrops();

    }

});

</script>
