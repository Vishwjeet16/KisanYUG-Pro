  <script>

async function checkLogin(){

  const response =
  await fetch("/check-login");

  const data =
  await response.json();

  const loginBtn =
  document.getElementById("loginBtn");

  const logoutBtn =
  document.getElementById("logoutBtn");

  if(data.loggedIn){

    loginBtn.classList.add("hidden");

    logoutBtn.classList.remove("hidden");

  }

  else{

    loginBtn.classList.remove("hidden");

    logoutBtn.classList.add("hidden");

  }

}

checkLogin();

const cartKey = "kisanYugCartItems";
const cartButton = document.getElementById("MEcart");
const cartDrawer = document.getElementById("cartDrawer");
const cartOverlay = document.getElementById("cartOverlay");
const closeCartButton = document.getElementById("closeCart");
const clearCartButton = document.getElementById("clearCart");
const cartItemsContainer = document.getElementById("cartItems");
const cartCount = document.getElementById("cartCount");
const cartTotal = document.getElementById("cartTotal");

function getCartItems(){

  return JSON.parse(
    localStorage.getItem(cartKey) || "[]"
  );

}

function saveCartItems(items){

  localStorage.setItem(
    cartKey,
    JSON.stringify(items)
  );

}

function getProductData(button){

  const card = button.closest(".product-card");
  const name = card.querySelector("h3")?.textContent.trim() || "Product";
  const farm = card.querySelector("p")?.textContent.trim() || "KisanYUG Market";
  const image = card.querySelector("img")?.src || "";
  const priceText = card.querySelector("h4")?.textContent.trim() || button.textContent.replace("Add To Cart -", "").trim();
  const price = Number(
    priceText.replace(/[^0-9.]/g, "")
  ) || 0;

  return {
    id: `${name}-${farm}`.toLowerCase().replace(/[^a-z0-9]+/g, "-"),
    name,
    farm,
    image,
    price,
    quantity: 1
  };

}

function renderCart(){

  const items = getCartItems();
  const totalItems = items.reduce((sum, item) => sum + item.quantity, 0);
  const totalPrice = items.reduce((sum, item) => sum + (item.price * item.quantity), 0);

  cartCount.textContent = totalItems;
  cartTotal.textContent = `$${totalPrice.toFixed(2)}`;

  if(items.length === 0){

    cartItemsContainer.innerHTML = `
      <div class="h-full flex flex-col items-center justify-center text-center text-gray-500 py-16">
        <iconify-icon icon="lucide:shopping-basket" class="text-5xl mb-3 text-gray-300"></iconify-icon>
        <p class="font-bold text-gray-700">Cart is empty</p>
        <p class="text-sm">Add products from the marketplace.</p>
      </div>
    `;

    return;

  }

  cartItemsContainer.innerHTML = items.map(item => `
    <div class="flex gap-4 border rounded-2xl p-3 bg-white shadow-sm">
      <img src="${item.image}" alt="${item.name}" class="w-20 h-20 rounded-xl object-cover">
      <div class="flex-1 min-w-0">
        <h3 class="font-bold text-gray-900 truncate">${item.name}</h3>
        <p class="text-sm text-gray-500 truncate">${item.farm}</p>
        <p class="font-bold text-[var(--moss)] mt-1">$${item.price.toFixed(2)}</p>
        <div class="flex items-center justify-between mt-3">
          <div class="flex items-center gap-2">
            <button type="button" class="cart-qty w-8 h-8 rounded-full border flex items-center justify-center" data-id="${item.id}" data-action="decrease">-</button>
            <span class="font-bold min-w-6 text-center">${item.quantity}</span>
            <button type="button" class="cart-qty w-8 h-8 rounded-full border flex items-center justify-center" data-id="${item.id}" data-action="increase">+</button>
          </div>
          <button type="button" class="remove-cart text-red-500 font-bold text-sm" data-id="${item.id}">Remove</button>
        </div>
      </div>
    </div>
  `).join("");

}

function openCart(){

  cartOverlay.classList.remove("hidden");
  cartDrawer.classList.remove("translate-x-full");

}

function closeCart(){

  cartDrawer.classList.add("translate-x-full");
  setTimeout(() => cartOverlay.classList.add("hidden"), 250);

}

function addToCart(button){

  const product = getProductData(button);
  const items = getCartItems();
  const existingItem = items.find(item => item.id === product.id);

  if(existingItem){

    existingItem.quantity += 1;

  }
  else{

    items.push(product);

  }

  saveCartItems(items);
  renderCart();
  openCart();

}

document.querySelectorAll(".product-card button").forEach(button => {

  if(button.textContent.includes("Add To Cart")){

    button.addEventListener("click", () => addToCart(button));

  }

});

cartButton.addEventListener("click", openCart);
closeCartButton.addEventListener("click", closeCart);
cartOverlay.addEventListener("click", closeCart);
clearCartButton.addEventListener("click", () => {

  saveCartItems([]);
  renderCart();

});

cartItemsContainer.addEventListener("click", event => {

  const quantityButton = event.target.closest(".cart-qty");
  const removeButton = event.target.closest(".remove-cart");
  const items = getCartItems();

  if(quantityButton){

    const item = items.find(cartItem => cartItem.id === quantityButton.dataset.id);

    if(item && quantityButton.dataset.action === "increase"){
      item.quantity += 1;
    }

    if(item && quantityButton.dataset.action === "decrease"){
      item.quantity -= 1;
    }

    saveCartItems(items.filter(cartItem => cartItem.quantity > 0));
    renderCart();

  }

  if(removeButton){

    saveCartItems(
      items.filter(cartItem => cartItem.id !== removeButton.dataset.id)
    );
    renderCart();

  }

});

renderCart();

</script>
