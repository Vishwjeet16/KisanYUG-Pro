  <script>
const apiBase = location.port === "5000" ? "" : "http://127.0.0.1:5000";
const sellForm = document.getElementById("sellForm");
const messageBox = document.getElementById("messageBox");

function showMessage(message, success){
  messageBox.textContent = message;
  messageBox.className = success
    ? "rounded-2xl p-4 text-sm font-semibold bg-green-50 text-green-700 border border-green-200"
    : "rounded-2xl p-4 text-sm font-semibold bg-red-50 text-red-700 border border-red-200";
}

async function loadMyPass(){
  try{
    const response = await fetch(`${apiBase}/api/my-pass`, {
      credentials:"include"
    });
    const data = await response.json();

    if(data.success){
      document.getElementById("myPassId").textContent = data.pass.custom_id;
      document.getElementById("farmerPass").value = data.pass.custom_id;
      document.getElementById("farmerName").value = data.pass.name || "";
      return;
    }

    document.getElementById("myPassId").textContent = "No pass found";
  }
  catch(error){
    document.getElementById("myPassId").textContent = "Backend not connected";
  }
}

sellForm.addEventListener("submit", async event => {
  event.preventDefault();

  const cropData = {
    farmerPass: document.getElementById("farmerPass").value.trim(),
    farmerName: document.getElementById("farmerName").value.trim(),
    cropName: document.getElementById("cropName").value.trim(),
    cropQty: document.getElementById("cropQty").value.trim(),
    cropPrice: document.getElementById("cropPrice").value.trim(),
    cropVillage: document.getElementById("cropVillage").value.trim(),
    cropImage: document.getElementById("cropImage").value.trim()
  };

  try{
    const response = await fetch(`${apiBase}/api/upload-crop`, {
      method:"POST",
      headers:{
        "Content-Type":"application/json"
      },
      credentials:"include",
      body:JSON.stringify(cropData)
    });

    const data = await response.json();
    showMessage(data.message || data.error, data.success);

    if(data.success){
      sellForm.reset();
      loadMyPass();
    }
  }
  catch(error){
    showMessage("Backend se connect nahi ho paya. Flask server start karein.", false);
  }
});

loadMyPass();
  </script>
